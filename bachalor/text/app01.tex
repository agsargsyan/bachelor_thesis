#новый экземпляр объекта Symulator

$ns namtrace-all $nf

#создание узлов и соединений
source "nodes.tcl"
#метрики TCP
source "TCP.tcl"

#визуализация в nam
source "nam.tcl"   		
#процедура finish
source "finish.tcl"                                                                         
#запуск программы
$ns run
\end{verbatim}
